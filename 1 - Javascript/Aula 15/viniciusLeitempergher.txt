    }

for (let i = 0; i < 3; i++) {
    maior[i] = []
    for (let j = 0; j < 3; j++) {
        (matriz1[i][j] > matriz2[i][j]) ? maior += matriz1[i][j] : maior += matriz2[i][j]
        maior += "\t"
    }
    maior += "\n"
}

// 6

let matriz = []

for (let i = 0; i < 3; i++) {
    matriz[i] = []
    matriz[i][0] = prompt("Digite o nome do aluno")
    let soma = 0
    for (let j = 0; j < 3; j++) {
        matriz[i][j + 1] = +prompt("Digite a nota")
        soma += matriz[i][j + 1]
    }
    let media = soma / 3
    matriz[i][4] = media;
    matriz[i][5] = (media < 6) ? "Reprovado" : "Aprovado"
}


let matrizTxt = ""
for (let i = 0; i < 3; i++) {
    for (let j = 0; j < 6; j++) {
        matrizTxt += matriz[i][j] + "\t"
    }
    matrizTxt += "\n"
}

console.log(matrizTxt)
